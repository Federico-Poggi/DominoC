tessera* stdTessere()
{
const int lunghezza = 21;
const int maxValue = 6;
tessera* std = (tessera*)malloc(sizeof(tessera) * lunghezza);
if (std == NULL)
{
error("Errore di allocazione di memoria");
exit(EXIT_FAILURE);
}
int i = 0;
for (int n1 = 1; n1 <= maxValue; ++n1)
{
for (int n2 = n1; n2 <= maxValue; ++n2)
{
std[i].num1 = n1;
std[i++].num2 = n2;
}
}
return std;
}
